Create:
local imageList = mane.graphics.newArrayImage({“icon.png”, “icon2.png”})
local image = group:newLayerImage(imageList, 1, 200, 200) -- newLayerImage(imageArray, layerindex, x, y, xScale, yScale, ox, oy)

Property:
x = x,
y = y,
image = imageArray, -- Array of images (loaded, clear the object to delete!).
layerindex = layerindex or 0, -- Index of the current picture
xScale = xScale or 1,
yScale = yScale or 1,
ox = ox or nil,
oy = oy or nil,
quad = nil,
_type = “newLayerImage”,
color = {1,1,1,1,1},
angle = 0,
isVisible = true,
group = self,
events = {
collision = {},
preCollision = {},
postCollision = {},
touch = {},
update = {}
}

Methods:
:setColor(r, g, b, a)
:translate(x, y) -- change x, y (changes, not sets)
:rotate(angle) -- change the rotation (changes, not sets)
:scale(x, y) -- change the scale of x, y (changes, not sets)
:moveToGroup(newGroup) -- Move the object to another group
:remove() -- Delete object

:addEvent(nameEvent, listener, ...)
:removeEvent(nameEvent, listener)
