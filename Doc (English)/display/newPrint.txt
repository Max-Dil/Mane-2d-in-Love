Create:
local text = group:newPrint(“Hello World!”, 100, 100) -- newPrint(text, font, x, y, fontSize)

Property:
text = text, -- text (text or table colored)
x = x,
y = y,
font = type(font) == “string” and mane.fonts[font] or love.graphics.getFont(), -- font (loaded automatically)
fontSize = fontSize or 20, -- font size (set via setFontSize method)
mode = “fill”,
_type = “newPrint”,
color = {1,1,1,1,1},
angle = 0,
xScale = 1,
yScale = 1,
isVisible = true,
group = self,
events = {
collision = {},
preCollision = {},
postCollision = {},
touch = {},
update = {}
}

Methods:
:setFontSize(fontSize)

:setColor(r, g, b, a)
:translate(x, y) -- change x, y (changes, not sets)
:rotate(angle) -- modify rotation (modifies, not sets)
:scale(x, y) -- change the scale of x, y (changes, not sets)
:moveToGroup(newGroup) -- Move the object to another group
:remove() -- Delete object

:addEvent(nameEvent, listener, ...)
:removeEvent(nameEvent, listener)
